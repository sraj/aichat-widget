📦 CDN Integration (Script Tag)

Add this to your HTML to embed the widget:

<!-- Add widget script -->
<script src="https://cdn.example.com/ai-chat-widget.js"></script>

<!-- Initialize widget -->
<script>
  const widget = AIChatWidget.init({
    apiKey: 'your-api-key',
    connection: {
      protocol: 'sse',
      baseUrl: 'https://api.example.com'
    },
    theme: {
      primaryColor: '#3b82f6'
    }
  });

  // Listen to events
  widget.on('message', (message) => {
    console.log('New message:', message);
  });
</script>

📦 NPM Integration

Install and use in your JavaScript/TypeScript project:

# Install via npm/pnpm
pnpm add @aichat-widget/widget

# Or via npm
npm install @aichat-widget/widget
import { init } from '@aichat-widget/widget';

const widget = init({
  apiKey: 'your-api-key',
  connection: {
    protocol: 'websocket',
    baseUrl: 'https://api.example.com'
  }
});

// Control the widget programmatically
widget.open();
widget.close();
widget.sendMessage('Hello!');

// Clean up when done
widget.destroy();
